Table Hecho_Estadia {
  id_estadia int [pk]
  id_propiedad int [ref: > Dim_Propiedad.id_propiedad]
  id_anfitrion int [ref: > Dim_Anfitrion.id_anfitrion]
  id_huesped int [ref: > Dim_Huesped.id_huesped]
  id_tiempo int [ref: > Dim_Tiempo.id_tiempo]
  id_calidad int [ref: > Dim_Calidad.id_calidad]
  ingreso_estimado decimal
  ocupacion_estimada decimal
}

Table Dim_Anfitrion {
  id_anfitrion int [pk]
  es_superhost boolean
  total_propiedades_host int
  identidad_verificada boolean
}

Table Dim_Propiedad {
  id_propiedad int [pk]
  tipo_propiedad varchar
  tipo_habitacion varchar
  capacidad int
  precio_noche decimal
  barrio varchar
  ciudad varchar
}

Table Dim_Huesped {
  id_huesped int [pk]
}

Table Dim_Tiempo {
  id_tiempo int [pk]
  fecha_resena date
  anio int
  mes int
  trimestre int
  temporada varchar
}

Table Dim_Calidad {
  id_calidad int [pk]
  puntaje_global int
  puntaje_precision int
  puntaje_limpieza int
  puntaje_checkin int
  puntaje_comunicacion int
  puntaje_ubicacion int
  puntaje_valor int
}
