// Use DBML to define your database structure

Table Anfitrion {
  id_anfitrion int [pk]
  es_superhost boolean
  total_propiedades_host int
  identidad_verificada boolean
}

Table Propiedad {
  id_propiedad int [pk]
  id_anfitrion int [ref: > Anfitrion.id_anfitrion]
  tipo_propiedad varchar
  tipo_habitacion varchar
  capacidad int
  n_habitaciones int
  precio_noche decimal
  min_noches int
  max_noches int
  barrio varchar
  ciudad varchar
  latitud decimal
  longitud decimal
}

Table Huesped {
  id_huesped int [pk]
}

Table Resena {
  id_resena int [pk]
  id_propiedad int [ref: > Propiedad.id_propiedad]
  id_huesped int [ref: > Huesped.id_huesped]
  fecha_resena date
}

Table Calificacion {
  id_calificacion int [pk]
  id_resena int [ref: > Resena.id_resena]
  puntaje_global int
  puntaje_precision int
  puntaje_limpieza int
  puntaje_checkin int
  puntaje_comunicacion int
  puntaje_ubicacion int
  puntaje_valor int
}
